137 template<typename T> void Statistics<T>::addElement(T element) {
138  double oldMean = mean;
139 
140  nrElements++;
141 
142  if ( nrElements == 1 ) {
143  mean = static_cast<double>(element);
144  harmonicMean = static_cast<double>(1.0 / element);
145  rms = static_cast<double>(element * element);
146  min = element;
147  max = element;
148  return;
149  }
150  mean = oldMean + ((element - oldMean) / nrElements);
151  harmonicMean += static_cast<double>(1.0 / element);
152  variance += (element - oldMean) * (element - mean);
153  rms += static_cast<double>(element * element);
154 
155  if ( element < min ) {
156  min = element;
157  } else if ( element > max ) {
158  max = element;
159  }
160 }
