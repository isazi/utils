141  for ( auto item = args.begin(); item != args.end(); ++item ) {
142  auto next = item;
143 
144  if (option == *item) {
145  std::advance(next, 1);
146  if ( next == args.end() ) {
147  throw SwitchNotFound(option);
148  }
149  retVal = isa::utils::castToType<std::string, T>(*next);
150  std::advance(next, 1);

153  }
