176 // Implementation from http://stackoverflow.com/questions/2782725/converting-float-values-from-big-endian-to-little-endian
177 inline void bigEndianToLittleEndian(char * value) {
178  unsigned int bitmap = *(reinterpret_cast<unsigned int *>(value));
179 
180  bitmap = ((bitmap >> 8) & 0x00ff00ff) | ((bitmap << 8) & 0xff00ff00);
181  bitmap = ((bitmap >> 16) & 0x0000ffff) | ((bitmap << 16) & 0xffff0000);
182 
183  *value = bitmap;
184 }
