178 // Implementation from http://stackoverflow.com/questions/2782725/converting-float-values-from-big-endian-to-little-endian
179 inline void bigEndianToLittleEndian(std::uint32_t * value) {
180  std::uint32_t bitmap = *value;
181 
182  bitmap = ((bitmap >> 8) & 0x00ff00ff) | ((bitmap << 8) & 0xff00ff00);
183  bitmap = ((bitmap >> 16) & 0x0000ffff) | ((bitmap << 16) & 0xffff0000);
184 
185  *value = bitmap;
186 }
